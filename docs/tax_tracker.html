348    def _open(self, date: pd.Timestamp, ticker: str, shares: float, price: float) -> Tuple[List[Lot], List[Lot]]:
349        """Process trade that opens or increases existing position in `ticker`.
350
351        Args:
352            date (pd.Timestamp): The date of the trade.
353            ticker (str): The asset ticker symbol.
354            shares (float): The number of shares traded.
355            price (float): The price at which the trade was executed.
356
357        Returns:
358            Tuple[List[Lot], List[Lot]]: A tuple containing:
359                1. List of new lots created during this trade.
360                2. List of closed lots created during this trade.
361        """
362        new_lots = []
363        closed_lots = []
364        # Check for loss trades on same side that would be washed by this
365        side = self.sign(shares)
366        loss_trades = self.loss_longs if (side > 0) else self.loss_shorts
367        while ticker in loss_trades and loss_trades[ticker] and abs(shares) > Config.MIN_SHARE_SIZE:
368            loss_trade = loss_trades[ticker].pop()
369            if (date - loss_trade.close_date).days <= WASH_SALE_WINDOW:
370                # This loss_trade washes a capital loss
371                washed_loss = loss_trade.tax_gain
372                if (side*shares) >= (side*loss_trade.shares + Config.MIN_SHARE_SIZE):
373                    # Entire loss-trade is washed
374                    washed_quantity = loss_trade.shares
375                else:
376                    # We only washed part of the loss; pro-rate it
377                    washed_quantity = shares
378                    washed_fraction = washed_quantity / loss_trade.shares
379                    washed_loss *= washed_fraction
380                    # Put the un-washed quantity back onto loss_trades
381                    unwashed_loss_trade = loss_trade.split(loss_trade.shares - washed_quantity)
382                    if abs(unwashed_loss_trade.shares) > Config.MIN_SHARE_SIZE:
383                        loss_trades[ticker].append(unwashed_loss_trade)
384                        closed_lots.append(unwashed_loss_trade)
385                        self.complete_trades[loss_trade.close_date].append(unwashed_loss_trade)
386                loss_trade.washed = True
387                # Reverse the wash trade
388                self.capital_gains[loss_trade.close_date][loss_trade.gain_type_verbose] -= washed_loss
389                # Buy the washed_quantity, but account for the washed trade's loss and effective purchase date
390                new_lots.append(Lot(ticker, washed_quantity, date, price, basis_add=washed_loss, basis_days=loss_trade.term))
391                shares -= washed_quantity
392                self.wash_trades.append(WashLoss(loss_trade, date, washed_quantity, washed_loss))
393        if abs(shares) > Config.MIN_SHARE_SIZE:
394            new_lots.append(Lot(ticker, shares, date, price))
395        self.positions[ticker].extend(new_lots)
396        self._last_date = date
397        return new_lots, closed_lots

399    def _close(self, date: pd.Timestamp, ticker: str, shares: float, price: float) -> Tuple[List[Lot], List[Lot]]:
400        """Process a trade that closes or reduces existing position in `ticker`.
401
402        Args:
403            date (pd.Timestamp): The date of the trade.
404            ticker (str): The asset ticker symbol.
405            shares (float): The number of shares traded.
406            price (float): The price at which the trade was executed.
407
408        Returns:
409            Tuple[List[Lot], List[Lot]]: A tuple containing:
410                1. List of new lots created during this trade.
411                2. List of closed lots created during this trade.
412        """
413        new_lots = []
414        closed_lots = []
415        side = -self.sign(shares)  # Side of existing position (which we are trading to close)
416        while abs(shares) > Config.MIN_SHARE_SIZE and self.positions[ticker]:
417            closing_lot = self.positions[ticker].popleft()
418            if (-side*shares + Config.MIN_SHARE_SIZE) < (side*closing_lot.shares):
419                # Partial close: Put unclosed quantity back in positions, pro-rating capital gains
420                unclosed_position = closing_lot.split(closing_lot.shares + shares)
421                if abs(unclosed_position.shares) > Config.MIN_SHARE_SIZE:
422                    new_lots.append(unclosed_position)
423                    self.positions[ticker].appendleft(unclosed_position)
424            closing_lot.close_px = price
425            closing_lot.close_date = date
426            closed_lots.append(closing_lot)
427
428            # If position.gain is negative, check whether this trade is a wash sale
429            gain = closing_lot.tax_gain  # Following logic is destructive, so work on a copy of the value
430            if gain < .0:
431                washable_shares = closing_lot.shares
432                washed_shares = .0
433                washed_gain = .0
434                for open_position in list(self.positions[ticker]):
435                    if abs(washable_shares) < Config.MIN_SHARE_SIZE:
436                        break
437                    # Find opening trades in this ticker in the last WASH_SALE_WINDOW days, made on days other than this trade's date:
438                    if (open_position.open_date != closing_lot.open_date) and (date - open_position.open_date).days <= WASH_SALE_WINDOW:
439                        assert self.sign(open_position.shares) == self.sign(closing_lot.shares), 'Washing opening trades must be same side'
440                        # This open washes the loss on current close; we have to adjust the opening trade's basis
441                        if (Config.MIN_SHARE_SIZE + abs(washable_shares)) < abs(open_position.shares):
442                            # We have to split the opening trade as it is only partially washed
443                            washed_proportion = washable_shares / open_position.shares
444                            washed_position = open_position.split(washable_shares)
445                            washed_position.basis_add += gain
446                            washed_position.basis_days += closing_lot.term
447                            # Create a new Lot for the washed quantity
448                            self.positions[ticker].append(washed_position)
449                            new_lots.append(washed_position)
450                            washed_shares += washable_shares
451                            washed_gain += washed_position.basis_add
452                            washable_shares = .0
453                            gain = .0
454                        else:  # open.shares <= washable_shares: This opening trade is fully washed
455                            washed_proportion = open_position.shares / washable_shares
456                            open_position.basis_add += gain * washed_proportion
457                            open_position.basis_days += closing_lot.term
458                            washed_shares += open_position.shares
459                            washed_gain += washed_proportion * gain
460                            washable_shares -= open_position.shares
461                            gain *= (1 - washed_proportion)
462                if washed_shares:
463                    closing_lot.washed = True
464                    self.wash_trades.append(WashLoss(closing_lot, date, washed_shares, washed_gain))
465
466            # Process capital gains
467            if gain != .0:
468                holding_period = (date - closing_lot.open_date).days + closing_lot.basis_days
469                if holding_period >= LONG_TERM_HOLDING_PERIOD:
470                    self.capital_gains[date]['LongTerm'] += gain
471                else:
472                    self.capital_gains[date]['ShortTerm'] += gain
473                if gain < .0:
474                    loss_trades = self.loss_longs if (side > 0) else self.loss_shorts
475                    loss_trades[ticker].append(closing_lot)
476
477            shares += closing_lot.shares
478
479        self._last_date = date
480        self.complete_trades[date].extend(closed_lots)
481        return new_lots, closed_lots

 684    def trade(self, date: pd.Timestamp, ticker: str, shares: float, price: float) -> Tuple[List[Lot], List[Lot]]:
 685        """Record a trade, processing as opening or closing based on existing position in ticker.
 686
 687        Args:
 688            date (pd.Timestamp): The date of the trade.
 689            ticker (str): The asset ticker symbol.
 690            shares (float): The number of shares traded.
 691            price (float): The price at which the trade was executed.
 692

 694            Tuple[List[Lot], List[Lot]]: A tuple containing:
 695                1. List of new lots created during this trade.
 696                2. List of closed lots created during this trade.

 698        if not self._last_BoD or (date > self._last_BoD):
 699            self.BoD(date)

 714    def receive_distribution(self, date: pd.Timestamp, ticker: str, dist_amount: float, dist_type: str):
 715        """Record a distribution for a ticker on a date.
 716
 717        Args:
 718            date (pd.Timestamp): The date of the distribution.
 719            ticker (str): The asset ticker symbol.
 720            dist_amount (float): The distribution amount per share.
 721            dist_type (str): The type of distribution.
 722
 723        Notes: Distributions are paid on holdings as of the end of the previous day.
 724        """
 725        if not self._last_BoD or (date > self._last_BoD):
 726            self.BoD(date)
 727        assert self._last_BoD == date, \
 728            f"Received distribution for {date} with last BoD {self._last_BoD}"
 729
 730        dist_type = dist_type[0].upper()
 731        shares = self.get_eod_shares(date - pd.Timedelta(days=1), ticker)
 732        if shares > 0:
 733            for position in self.positions[ticker]:
 734                position.distributions += position.shares * dist_amount
 735            if dist_type == 'L':
 736                self.daily_dividends[date]['LongTermGain'] += shares * dist_amount
 737            elif dist_type == 'S':
 738                self.daily_dividends[date]['ShortTermGain'] += shares * dist_amount
 739            elif dist_type in ('E', 'R'):  # Tax-exempt or Return of Capital
 740                self.daily_dividends[date]['Exempt'] += shares * dist_amount
 741            elif dist_type == 'N':  # Not qualified
 742                self.daily_dividends[date]['Regular'] += shares * dist_amount
 743            # For dividends that can become qualified, we add them as 'Regular' to begin with
 744            # and later move them to 'Qualified' if we subsequently find the holding period is satisfied
 745            elif dist_type == 'D':  # Regular dividend
 746                self.daily_dividends[date]['Regular'] += shares * dist_amount
 747                for position in self.positions[ticker]:
 748                    self.pending_qualified.append((date, position, dist_amount))
 749            elif dist_type == 'P':  # Preferred dividend
 750                self.daily_dividends[date]['Regular'] += shares * dist_amount
 751                for position in self.positions[ticker]:
 752                    self.pending_qualified_preferred.append((date, position, dist_amount))
 753            else:
 754                raise ValueError(f"Unknown distribution type {dist_type}")
 755
 756        elif shares < 0:  # Short positions make payments in lieu of dividends
 757            for position in self.positions[ticker]:
 758                position.distributions += position.shares * dist_amount
 759                # After 45 days, payments in lieu of dividends on short positions are interest expenses.
 760                # Ref IRS Publication 550 p.86.
 761                if (date - position.open_date).days > PIL_CONVERSION_PERIOD:
 762                    self.daily_dividends[date]['Expense'] += position.shares * dist_amount
 763                else:
 764                    # Log the payment in lieu to the position; to be accounted
 765                    #  when the position is closed or held for more than PIL_CONVERSION_PERIOD.
 766                    if not position.payment_in_lieu:
 767                        self.pending_pil_positions[position.open_date].append(position)
 768                    position.payment_in_lieu[date] += dist_amount
 769                    self.daily_dividends[date]['PayInLieu'] += position.shares * dist_amount

 771    def count_qualified_shares(self, ticker: str, exdate: pd.Timestamp, period: int) -> float:
 772        """Return the number of shares of `ticker` that were held for at least `period`+1 days
 773            when looking at the date range of `period` days before and after `exdate`.
 774
 775        Args:
 776            ticker (str): The asset ticker symbol.
 777            exdate (pd.Timestamp): The ex-dividend date.
 778            period (int): The limiting holding period in days before qualification.
 779
 780        Returns:
 781            float: The number of shares that were held for at least `period`+1 days and not already qualified.
 782        """
 783        q = self.holdings.shift().loc[(exdate - pd.Timedelta(days=period)):(exdate + pd.Timedelta(days=period)),
 784                                      ticker].fillna(0).nlargest(period+1)

 786            return 0
 787        return q.min() - self.already_qualified[(exdate, ticker)]

 828    def BoD(self, date: pd.Timestamp) -> Dict[str, float]:
 829        """Beginning of Day (BoD) processing for a given date. This method must be called before 
 830            any transactions are entered on the specified date to ensure that incoming positions 
 831            going ex-dividend are properly accounted for.
 832
 833        Args:
 834            date (pd.Timestamp): The date for which the BoD processing is being performed.
 835
 836        Returns:
 837            Dict[str, float]: A dictionary keyed by ticker, listing total dividends received per share
 838                  for all tickers held on the specified date.
 839
 840        Raises:
 841            AssertionError: If the method is called with a date not later than the last BoD date.
 842
 843        Notes:
 844            - Ensures that the holdings DataFrame is extended to include the specified date.
 845            - Fills in skipped days by iteratively calling BoD for intermediate dates.
 846            - Calls End of Day (EoD) processing for the previous day if it was not already called.
 847            - Processes distributions for stocks held on the specified date.
 848            - Reviews pending distributions to determine if they qualify as "qualified dividends" 
 849              based on IRS holding period rules.
 850            - Handles pending payment-in-lieu (PIL) positions to account for interest expenses 
 851              on short positions held for more than PIL_CONVERSION_PERIOD days.
 852        """
 853        assert (not self._last_BoD) or date > self._last_BoD, \
 854            f"BoD called {date} with last BoD={self._last_BoD}"
 855
 856        # Make sure holdings extends to current date:
 857        if not date in self.holdings.index:
 858            self.holdings = self.holdings.reindex(self.holdings.index.union(
 859                pd.date_range(start=min(date, self.holdings.index.min() or date),
 860                              end=max(date, self.holdings.index.max() or date)))).fillna(.0)
 861
 862        # If we skipped any days, go back and fill them in:
 863        if (last_date := self._last_BoD) and (date - last_date).days > 1:
 864            for d in pd.date_range(start=last_date, end=date, inclusive='neither'):
 865                self.BoD(d)
 866        # If EoD wasn't called for previous day then call it:
 867        if self._last_BoD and ((not self._last_EoD) or (date - self._last_EoD).days > 1):
 868            self.EoD(self._last_BoD)
 869        self._last_BoD = date
 870
 871        # Receive distributions for this exdate:
 872        dividends_received: Dict[str, float] = {}
 873        if self.distributions is not None:
 874            for ticker, group in self.distributions[self.distributions.index.get_level_values('Date') == date].groupby(level='Ticker'):
 875                if self.get_position(ticker):
 876                    for idx, row in group.iterrows():
 877                        dist_type = idx[2]
 878                        dist_amount = row['Distribution']
 879                        self.receive_distribution(date, ticker, dist_amount, dist_type)
 880
 881                    if ticker not in dividends_received:
 882                        dividends_received[ticker] = 0.0
 883                    dividends_received[ticker] += dist_amount
 884
 885        #region Review pending distributions to see if any have become qualified due to holding period
 886        '''From IRS Publication 550, p.28 – HOLDING PERIOD:
 887            You must have held the stock for more than 60 days during the 121-day period that begins
 888                60 days before the ex-dividend date. When counting the number of days you held the stock,
 889                include the day you disposed of the stock, but not the day you acquired it.
 890            In the case of preferred stock, you must have held the stock more than 90 days during the
 891                181-day period that begins 90 days before the ex-dividend date if the dividends are
 892                due to periods totaling more than 366 days.
 893        '''
 894        for pending in list(self.pending_qualified):
 895            exdate, position, dist_amount = pending
 896            if (date - exdate).days > QUALIFIED_WINDOW:
 897                # If not qualified by now then it never will be
 898                self.pending_qualified.remove(pending)
 899            elif (qualified_shares := self.count_qualified_shares(position.ticker, exdate, QUALIFIED_WINDOW)) > 0:
 900                # Number of shares held at the close of the day before the ex-dividend date
 901                dividend_shares = self.get_eod_shares(exdate-pd.Timedelta(days=1), position.ticker)
 902                assert dividend_shares >= qualified_shares
 903                self.already_qualified[(exdate, position.ticker)] += qualified_shares
 904                self.daily_dividends[exdate]['Qualified'] += qualified_shares * dist_amount
 905                self.daily_dividends[exdate]['Regular'] -= qualified_shares * dist_amount
 906                self.pending_qualified.remove(pending)
 907
 908        for pending in list(self.pending_qualified_preferred):
 909            exdate, position, dist_amount = pending
 910            if (date - exdate).days > PREFERRED_QUALIFIED_WINDOW:
 911                # If not qualified by now then it never will be
 912                self.pending_qualified_preferred.remove(pending)
 913            elif (qualified_shares := self.count_qualified_shares(position.ticker, exdate, PREFERRED_QUALIFIED_WINDOW)) > 0:
 914                # Number of shares held at the close of the day before the ex-dividend date
 915                dividend_shares = self.get_eod_shares(exdate-pd.Timedelta(days=1), position.ticker)
 916                assert dividend_shares >= qualified_shares
 917                self.already_qualified[(exdate, position.ticker)] += qualified_shares
 918                self.daily_dividends[exdate]['Qualified'] += qualified_shares * dist_amount
 919                self.daily_dividends[exdate]['Regular'] -= qualified_shares * dist_amount
 920                self.pending_qualified_preferred.remove(pending)
 921
 922        # Review pending_pil_positions to see if any have become interest expenses
 923        #  because the short position was open for more than PIL_CONVERSION_PERIOD:
 924        for open_date in sorted(self.pending_pil_positions.keys()):
 925            if (date - open_date).days <= PIL_CONVERSION_PERIOD:
 926                break
 927            for position in self.pending_pil_positions[open_date]:
 928                assert position.is_open
 929                assert position.shares < 0
 930                for dist_date, dist_amount in position.payment_in_lieu.items():
 931                    pil = position.shares * dist_amount
 932                    self.daily_dividends[dist_date]['PayInLieu'] -= pil
 933                    self.daily_dividends[dist_date]['Expense'] += pil
 934                position.payment_in_lieu.clear()
 935            del self.pending_pil_positions[open_date]
 936        #endregion Review pending distributions
 937        return dividends_received
